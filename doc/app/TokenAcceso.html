def self.generar(juego_id, usuario_id)

  token = TokenAcceso.new
  token.juego_id = juego_id
  token.usuario_id = usuario_id

  r = Random.new
  token.token = Array.new(250) { CARACTERES[r.rand(CARACTERES.length)] }.join
  
  token
  
end
